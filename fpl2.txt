	int y     0

	int wait  5

	cmd <arm>

	nop wait
	cmd <move %x %y %z>
	add wait 5
	nop wait

	cmd <move %x %y %z>
	add wait 5
	nop wait

	set x -50
	set y -50
	set z -50
	cmd <move %x %y %z>
	add wait 5
	nop wait

	set y -50
	set z 50
	cmd <move %x %y %z>
	add wait 5
	nop wait

	set y 50
	set z -50

	nop wait

	bne move1
